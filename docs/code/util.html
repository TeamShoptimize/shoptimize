window.findMinIndex = (array) ->
    index = 0
    curMin = Number.POSITIVE_INFINITY
    curMinIndex = -1
    for elem in array
        if elem < curMin
            curMin = elem
            curMinIndex = index
        index++
    return curMinIndex

window.zero2D = (rows, cols) ->
  array = []
  row = []
  while (cols--)
    row.push(0)
  while (rows--)
    array.push(row.slice())
  array
